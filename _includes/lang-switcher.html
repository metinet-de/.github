{% comment %} Language Switcher ───────────────────────────────────────── Finds the equivalent page/post in the other language using the `ref` front-matter key. Falls back to the other language's home page if no translation exists. Outputs a single link: "Deutsch" or "English". {% endcomment %} {%- assign current_lang = page.lang | default: "en" -%} {%- assign t = site.data.i18n[current_lang] -%} {%- assign other_lang = t.switch_lang_code -%} {%- assign other_label = t.switch_language -%} {%- comment -%} Try to find a matching translation via `ref` {%- endcomment -%} {%- assign translation_url = nil -%} {%- if page.ref -%} {%- comment -%} Search pages first {%- endcomment -%} {%- for pg in site.pages -%} {%- if pg.ref == page.ref and pg.lang == other_lang -%} {%- assign translation_url = pg.url | relative_url -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- comment -%} Then search posts {%- endcomment -%} {%- unless translation_url -%} {%- for pt in site.posts -%} {%- if pt.ref == page.ref and pt.lang == other_lang -%} {%- assign translation_url = pt.url | relative_url -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- endunless -%} {%- endif -%} {%- comment -%} Fallback: link to the other language's home page {%- endcomment -%} {%- unless translation_url -%} {%- if other_lang == "de" -%} {%- assign translation_url = "/de/" | relative_url -%} {%- else -%} {%- assign translation_url = "/" | relative_url -%} {%- endif -%} {%- endunless -%} {{ other_label }}
