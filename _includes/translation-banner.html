{% comment %} Translation Banner ───────────────────────────────────────── Shows a small banner at the top of a blog post linking to its translation in the other language, if one exists. Only renders when a translated counterpart is found via `ref`. {% endcomment %} {%- assign current_lang = page.lang | default: "en" -%} {%- assign t = site.data.i18n[current_lang] -%} {%- assign other_lang = t.switch_lang_code -%} {%- assign other_name = t.switch_language -%} {%- if page.ref -%} {%- assign translated_post = nil -%} {%- for pt in site.posts -%} {%- if pt.ref == page.ref and pt.lang == other_lang -%} {%- assign translated_post = pt -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- unless translated_post -%} {%- for pg in site.pages -%} {%- if pg.ref == page.ref and pg.lang == other_lang -%} {%- assign translated_post = pg -%} {%- break -%} {%- endif -%} {%- endfor -%} {%- endunless -%} {%- if translated_post -%}
{{ t.post.also_in }} {{ other_name }}: {{ translated_post.title }}
{%- endif -%} {%- endif -%}
